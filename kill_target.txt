player = find.allPlayers(function(ply) 
    return string.find(ply:getName():lower(), ("lil"):lower(), 1, false)
end)[1]
hook.add("think", "hurt", function() 
   
    if owner():keyDown(IN_KEY.RELOAD) then
        game.blastDamage(player:obbCenterW(), 1000, 10000)
    end
end)
